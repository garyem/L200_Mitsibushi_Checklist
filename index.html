function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const data = JSON.parse(e.postData.contents); // Write headers if sheet is empty if (sheet.getLastRow() === 0) { const headers = [ 'Timestamp', 'Reg Plate', 'Year/Plate', 'Asking Price (£)', 'Mileage', 'Trim', 'Gearbox', 'Dealer/Location', 'Colour', 'Tonneau Fitted', 'Overall Notes', 'Would Buy', 'Total Passed', 'Total Flagged', 'Verdict', // Paperwork 'V5C Logbook', 'Full Service History', 'Warranty Transferable', 'No Outstanding Finance', 'No Outstanding Recalls', 'MOT History Checked', // Exterior 'Panel Gaps Even', 'No Chassis Rust', 'Load Bed OK', 'Tyres OK', 'Lights & Camera Work', // Under Bonnet 'Cold Start OK', 'Oil Level OK', 'Coolant OK', 'No Oil Leaks', 'Turbo Healthy', 'No Warning Lights', // Test Drive 'No Vibration 50-70mph', 'Gearbox Smooth', 'Steering Straight', 'Brakes OK', '4WD System OK', 'Air Con Cold', // Interior 'Infotainment Works', 'Windows Work', 'Seats & Trim OK', 'No Damp/Smell', // Price 'Price In Range', 'Day-One Budget OK', 'Inspection Booked' ]; sheet.appendRow(headers); // Style the header row const headerRange = sheet.getRange(1, 1, 1, headers.length); headerRange.setBackground('#1a1a1a'); headerRange.setFontColor('#d0a020'); headerRange.setFontWeight('bold'); headerRange.setFrozenRows(1); } // Parse individual checklist items const checklist = data.checklistItems || {}; const row = [ data.timestamp, data.regPlate, data.yearPlate, data.askingPrice, data.mileage, data.trimLevel, data.gearbox, data.dealerLocation, data.colour, data.tonneauFitted, data.overallNotes, data.wouldBuy, data.passed, data.flagged, data.verdict, // Paperwork checklist['v5c'] || 'Not checked', checklist['serviceHistory'] || 'Not checked', checklist['warranty'] || 'Not checked', checklist['hpi'] || 'Not checked', checklist['recalls'] || 'Not checked', checklist['mot'] || 'Not checked', // Exterior checklist['panelGaps'] || 'Not checked', checklist['chassisRust'] || 'Not checked', checklist['loadBed'] || 'Not checked', checklist['tyres'] || 'Not checked', checklist['lights'] || 'Not checked', // Under Bonnet checklist['coldStart'] || 'Not checked', checklist['oilLevel'] || 'Not checked', checklist['coolant'] || 'Not checked', checklist['oilLeaks'] || 'Not checked', checklist['turbo'] || 'Not checked', checklist['warningLights'] || 'Not checked', // Test Drive checklist['vibration'] || 'Not checked', checklist['gearbox'] || 'Not checked', checklist['steering'] || 'Not checked', checklist['brakes'] || 'Not checked', checklist['fourWD'] || 'Not checked', checklist['airCon'] || 'Not checked', // Interior checklist['infotainment'] || 'Not checked', checklist['windows'] || 'Not checked', checklist['seats'] || 'Not checked', checklist['damp'] || 'Not checked', // Price checklist['priceRange'] || 'Not checked', checklist['dayOneCosts'] || 'Not checked', checklist['inspection'] || 'Not checked' ]; sheet.appendRow(row); // Colour code the new row based on result const newRow = sheet.getLastRow(); const checkStart = 16; // Column P onwards (checklist items) const totalCols = row.length; for (let col = checkStart; col <= totalCols; col++) { const cell = sheet.getRange(newRow, col); const val = cell.getValue(); if (val === 'Passed') { cell.setBackground('#1e3328').setFontColor('#4caf6e'); } else if (val === 'ISSUE FLAGGED') { cell.setBackground('#331818').setFontColor('#e05252'); } else { cell.setBackground('#222222').setFontColor('#888888'); } } // Colour the Would Buy cell const wouldBuyCell = sheet.getRange(newRow, 12); const wouldBuyVal = wouldBuyCell.getValue(); if (wouldBuyVal.includes('Yes')) { wouldBuyCell.setBackground('#1e3328').setFontColor('#4caf6e'); } else if (wouldBuyVal.includes('No')) { wouldBuyCell.setBackground('#331818').setFontColor('#e05252'); } else { wouldBuyCell.setBackground('#332500').setFontColor('#f0a500'); } return ContentService.createTextOutput('OK'); }
